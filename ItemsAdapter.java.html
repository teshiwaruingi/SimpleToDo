package com.example.simpletodo;

import android.support.annotation.NonNull;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;

import androidx.annotation.NoNull;
import andriodx.reyclerview.widget.RecyclerView;

//Responsidble for displlaying data from the model into a row in the recyller view
public class ItemsAdapter extends RecyclerView.ViewHolder<ItemsAdapter.ViewHolder> {

    List<String> items;

    public ItemsAdapter(List<String> items) {
        this.items = items;
    }

    @NonNull
    @Override
    public ViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType){
        // Use layout infloatro to inflate view
        View todoView = LayoutInflater.from(parent.getContext()).inflate(android.R.layout.simple_list_item_1, parent, attachToRoot: false)
        // Wrap it inside a view holder and return it
        return new ViewHolder(todoView);
    }
    // responidlle for binding date to a particular view holder
    @Override
    public void onBindViewHolder(@NoNull ViewHolder holder, int postion){
        //Grab the intem at the postion
        String item = items.get(postion);
        // Bind the item into the specifed view holder
        holder.bind(item);


    }

    @Override
    public int getItemCount() {
        return items.size();
    }

    // Container to provide easy acces to viess that reprents each row of the llist
    class ViewHolder extends RecyclerView.ViewHolder {
        TextView tvItem;

        public ViewHolder (@NoNull View itemView){
            super(itemView);
            tvItem = itemView.findViewById(.android.R.id.text1);

        }
        //Update the view inside of the view holldr iwht this data
        public void bind(String item) {
            tvItem.setText(item);
        }
    }
}
